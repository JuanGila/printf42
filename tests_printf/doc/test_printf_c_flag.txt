//// CHAR TESTS

1.- Variable Char
    Descripcion: Se comprueba que se muestra un char normal.
    OK: Se muestra correctamente un char normal.
    KO: No se muestra correctamente un char normal.
    CRASH:
        Sigsev en caso de ¿?
        Sirargv en caso de ¿?

// Se comprueba que se muestra un char normal
ft_printf_res = ft_printf("Hello %c\n", '!');
printf_res = printf("Hello %c\n", '!');
ft_check_printf_res("Test1(normal-char)", printf_res, ft_printf_res);//test -> OK

// Se comprueba que se muestra un char negativo
ft_printf_res = ft_printf("Hello %c\n", -1);
printf_res = printf("Hello %c\n", -1);
ft_check_printf_res("Test2(negative-char)", printf_res, ft_printf_res);

// Se comprueba que se muestra un char 0
ft_printf_res = ft_printf("Hello %c\n", '\0');//ft_printf_res = ft_printf("Hello %c\n", 0);probar
printf_res = printf("Hello %c\n", '\0');//printf_res = printf("Hello %c\n", 0);probar
ft_check_printf_res("Test3(zero-char)", printf_res, ft_printf_res);

// Se comprueba que se muestran varios chars consecutivos.
ft_printf_res = ft_printf("Hello %c %c %c %c\n", 'a', 'b', 'c', 'd');
printf_res = printf("Hello %c %c %c %c\n", 'a', 'b', 'c', 'd');
ft_check_printf_res("Test4(consecutive-chars)", printf_res, ft_printf_res);

// Se comprueba que se muestran varios chars combinados.
ft_printf_res = ft_printf("Hello %c%c%c%c\n", 'a', '\0', 'c', '\0', '.');
printf_res = printf("Hello %c%c%c%c\n", 'a', '\0', 'c', '\0', '.');
ft_check_printf_res("Test5(combined-chars)", printf_res, ft_printf_res);
